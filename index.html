Blinking an LED using GPIO

import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
GPIO.setup(17, GPIO.OUT)

try:
    while True:
        GPIO.output(17, GPIO.HIGH)  # LED on
        time.sleep(1)
        GPIO.output(17, GPIO.LOW)   # LED off
        time.sleep(1)
except KeyboardInterrupt:
    GPIO.cleanup()
