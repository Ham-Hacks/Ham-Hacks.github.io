NodeMCU with Built-in OLED

Neat little NodeMCU ESP8266 board with an OLED baked in ‚Äî no jumper wires, just vibes. Here's how to make it work.

OLED Pinout:

  • SDA = GPIO12 (D6)
  • SCL = GPIO14 (D5)

Address: 0x3C (run scanner below to verify)

I2C Scanner
#include <Wire.h>

void setup() {
  Wire.begin(12, 14); // SDA, SCL
  Serial.begin(115200);
  Serial.println("Scanning for I2C devices...");
  for (byte i = 1; i < 127; i++) {
    Wire.beginTransmission(i);
    if (Wire.endTransmission() == 0) {
      Serial.print("Found device at 0x");
      Serial.println(i, HEX);
      delay(500);
    }
  }
}

void loop() {}
Hello OLED Example
#include <Wire.h>
#include <U8g2lib.h>

U8G2_SSD1306_128X64_NONAME_F_HW_I2C u8g2(
  U8G2_R0,      // rotation
  U8X8_PIN_NONE,// reset
  14,           // clock (D5)
  12            // data  (D6)
);

void setup() {
  u8g2.begin();
}

void loop() {
  u8g2.clearBuffer();
  u8g2.setFont(u8g2_font_ncenB08_tr);
  u8g2.drawStr(0, 24, "Hello from HamHacks!");
  u8g2.sendBuffer();
  delay(1000);
}
